const textDisplay = document.getElementById("text");
const phrases = [
  "Hello, my name is Dave ",
  "I love to code ",
  "I like to teach coding",
];

let i = 0;
let j = 0;

function loop() {
  if (i < phrases.length) {
    //console.log(phrases[i]);

    if (j < phrases[i].length) {
      console.log(phrases[i][j]);
      j++;

      if (j === phrases[i].length) {
        i++;
        j = 0;
      }
    }
  }

  setInterval(loop, 500);
}

loop();
